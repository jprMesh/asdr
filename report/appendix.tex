\section{Matched Filter Example}

\begin{verbatim}
clear all;
close all;

toMatch = [1 -1 -1 -1 1 1 -1 1];
noise = randn(400,1);
noise_signal = noise;

for i=50:length(toMatch) + 50
    noise_signal = noise(i) + toMatch(i-49);
end

\end{verbatim}
